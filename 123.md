#include <stdio.h>
#include <stdlib.h>
#include <math.h>
# define Ep 0.00001

int main(int argc, char** argv) {
	double dA=0.0,dB=0.0,dC=0.0;//Ax^2+Bx+c=0
	double dV=0.0;
	double dAns=0.0;
	printf("請輸入一個浮點數");
	scanf("%lf",&dA);
	printf("請輸入一個浮點數");
	scanf("%lf",&dB);
	printf("請輸入一個浮點數");
	scanf("%lf",&dC);
	printf("解一元二次方程式:%fx^2 + %fx + %f=0\n",dA,dB,dC);
	
	dV = dB*dB- 4.0*dA*dC;
	
	if( dV<0.0 ){
		printf("此式無實數解\n");
	} else if(-Ep < dV&&dV < Ep){
		dAns = -dB/(2.0*dA);
		printf("此式有重根:%f\n",dAns);
	} else {
		dAns = (-dB + sqrt(dV))/(2.0*dA);
		printf("此式有兩個根:%f 和",dAns);
		dAns = (-dB - sqrt(dV))/(2.0*dA);
		printf("%f\n",dAns);
	}
	return 0;
}
